namespace commutative_algebra

# Type definitions
def Ring : Type
def Module (R : Ring) : Type

# Constructions
def RingAsModule (R : Ring) : Module R

# Ring properties
def field artin r_noetherian r_local gorenstein euclidean r_reduced (R : Ring) : Prop
def ideal (R : Ring) : Type
def maximal_ideal prime_ideal radical_ideal {R : Ring} (I : ideal R) : Prop
def nilradical zero_ideal unit_ideal jacobson_radical (R : Ring) : ideal R

# Examples
def ZZ QQ RR CC: Ring

def QQIsField : field QQ
def RRIsField : field RR
def CCIsField : field CC

end commutative_algebra
